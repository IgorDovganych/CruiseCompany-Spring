	CREATE TABLE IF NOT EXISTS excursions(
		id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
		name VARCHAR(45) NOT NULL,
		port_id INT NOT NULL,
		price INT NOT NULL,
		description TEXT,
		isActive BOOLEAN,
		
		CONSTRAINT port_id_excursions
		FOREIGN KEY (port_id)
		REFERENCES ports(id)
	);
